        function analyzeImageBrightness(imageData) {
            const data = imageData.data;
            const width = imageData.width;
            const height = imageData.height;
            
            let topBrightness = 0, bottomBrightness = 0;
            let leftBrightness = 0, rightBrightness = 0;
            let topCount = 0, bottomCount = 0, leftCount = 0, rightCount = 0;
            
            for (let y = 0; y < height; y++) {
                for (let x = 0; x < width; x++) {
                    const idx = (y * width + x) * 4;
                    const brightness = (data[idx] + data[idx + 1] + data[idx + 2]) / 3;
                    
                    if (y < height / 2) {
                        topBrightness += brightness;
                        topCount++;
                    } else {
                        bottomBrightness += brightness;
                        bottomCount++;
                    }
                    
                    if (x < width / 2) {
                        leftBrightness += brightness;
                        leftCount++;
                    } else {
                        rightBrightness += brightness;
                        rightCount++;
                    }
                }
            }
            
            return {
                top: topBrightness / topCount,
                bottom: bottomBrightness / bottomCount,
                left: leftBrightness / leftCount,
                right: rightBrightness / rightCount
            };
